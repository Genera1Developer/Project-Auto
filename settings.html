// Error handling for input validation const form = document.querySelector('form'); form.addEventListener('submit', (e) => { e.preventDefault(); const proxyEnabled = document.querySelector('#proxy-enabled'); const proxyAddress = document.querySelector('#proxy-address'); const proxyPort = document.querySelector('#proxy-port'); const bandwidthLimitEnabled = document.querySelector('#bandwidth-limit-enabled'); const bandwidthLimit = document.querySelector('#bandwidth-limit'); // Check if proxy is enabled if (!proxyEnabled.checked) { alert('Proxy must be enabled'); return; } // Check if proxy address is valid if (!proxyAddress.value) { alert('Proxy address cannot be empty'); return; } // Check if proxy port is valid if (!proxyPort.value || isNaN(proxyPort.value)) { alert('Proxy port must be a valid number'); return; } // Check if bandwidth limit is enabled and valid if (bandwidthLimitEnabled.checked) { if (!bandwidthLimit.value || isNaN(bandwidthLimit.value)) { alert('Bandwidth limit must be a valid number'); return; } } // Submit form if all inputs are valid form.submit(); }); ```
