Web Proxy

``` **src/index.js** ```javascript import { createServer } from 'http'; import getOriginalUrl from './utils/get-original-url.js'; import getHostname from './utils/get-hostname.js'; const server = createServer(async (req, res) => { try { if (req.method === 'POST') { const { url } = await req.json(); const originalUrl = getOriginalUrl(url); const hostname = getHostname(originalUrl); // Perform the proxy request const proxyRes = await fetch(originalUrl, { headers: { 'x-forwarded-host': hostname, }, }); // Set the appropriate headers res.writeHead(proxyRes.status, proxyRes.headers); // Pipe the response to the client proxyRes.body.pipe(res); } else { res.writeHead(405); res.end('Method Not Allowed'); } } catch (err) { res.writeHead(500); res.end('Error: ' + err.message); } }); server.listen(3000, () => { console.log('Web proxy server listening on port 3000'); }); ``` **utils/get-original-url.js** ```javascript export default function getOriginalUrl(url) { if (url.startsWith('https://')) { return url.slice(8); } else if (url.startsWith('http://')) { return url.slice(7); } else { return url; } } ``` **utils/get-hostname.js** ```javascript export default function getHostname(url) { const parts = url.split('/'); return parts[0].replace(':', ''); } ``` **Changes:** * Updated the front-end to use a POST request instead of a GET request. * Added utility functions to extract the original URL and hostname from the request. * Fixed the issue where the proxy headers were not being set correctly.
