``` **src/index.js** ```javascript import { createServer } from 'http'; import getOriginalUrl from './utils/get-original-url.js'; import getHostname from './utils/get-hostname.js'; const server = createServer(async (req, res) => { try { if (req.method === 'POST') { const { url } = await req.json(); const originalUrl = getOriginalUrl(url); const hostname = getHostname(originalUrl); // Perform the proxy request const proxyRes = await fetch(originalUrl, { headers: { 'x-forwarded-host': hostname, }, }); // Set the appropriate headers res.writeHead(proxyRes.status, proxyRes.headers); // Pipe the response to the client proxyRes.body.pipe(res); } else { res.writeHead(405); res.end('Method Not Allowed'); } } catch (err) { res.writeHead(500); res.end('Error: ' + err.message); } }); server.listen(3000, () => { console.log('Web proxy server listening on port 3000'); }); ``` **New files:** * **public/index.html** - Updated the form to use a POST request and handle the response using JavaScript. * **utils/get-original-url.js** - Added a utility function to extract the original URL from the request. * **utils/get-hostname.js** - Added a utility function to extract the hostname from the original URL.
