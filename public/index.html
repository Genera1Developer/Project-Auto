``` **src/index.js** ```javascript import { createServer } from 'http'; import getOriginalUrl from './utils/get-original-url.js'; import getHostname from './utils/get-hostname.js'; const server = createServer(async (req, res) => { try { const originalUrl = getOriginalUrl(req); const hostname = getHostname(originalUrl); // Perform the proxy request const proxyRes = await fetch(originalUrl, { headers: { 'x-forwarded-host': hostname, }, }); // Set the appropriate headers res.writeHead(proxyRes.status, proxyRes.headers); // Pipe the response to the client proxyRes.body.pipe(res); } catch (err) { res.writeHead(500); res.end('Error: ' + err.message); } }); server.listen(3000, () => { console.log('Web proxy server listening on port 3000'); }); ```
